const formAdicionarPlano = document.getElementById('form-adicionar-plano');

formAdicionarPlano.addEventListener('submit', function(event) {
    event.preventDefault();

    const tipo = document.getElementById('tipo').value;
    const visa = document.getElementById('visa').value;
    // Obtenha os outros valores do formulário

    const novoPlano = {
        tipo: tipo,
        visa: visa,
        // Adicione os outros campos
    };

    planos.push(novoPlano);
    atualizarTabela(); // Função para atualizar a tabela com os novos dados
    formAdicionarPlano.reset(); // Limpa o formulário
});
